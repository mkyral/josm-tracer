021package org.openstreetmap.josm.plugins.tracer.connectways;
022
023public interface IDiscardableCutoffPredicate {
024
025    /**
026     * Replies whether the given way can be completely removed as a result of polygon clipping.
027     * If <code>cutoffs_percent</code> is zero, entire way was clipped out.
028     * If <code>cutoffs_percent</code> is non-zero, way was cut to one or more
029     * cutoffs with areas below the given threshold. (Note that the sum of all
030     * cutoffs can be greater than the threshold.)
031     * @param way way to be discarded
032     * @param cutoffs_percent sum of all cutoffs, expressed in percent of the original area
033     * @return true if way can be discarded, false otherwise
034     */
035    public boolean canSilentlyDiscard(EdWay way, double cutoffs_percent);
036}

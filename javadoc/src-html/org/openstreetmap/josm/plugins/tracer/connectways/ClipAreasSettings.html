020package org.openstreetmap.josm.plugins.tracer.connectways;
021
022public class ClipAreasSettings {
023    private final GeomDeviation m_connectSubjectToClipTolerance;
024    private final GeomDeviation m_clipperWayCleanupsTolerance;
025    private final GeomDeviation m_reconnectIntersectionNodesTolerance;
026    private final double m_discardCutoffsPercent;
027    private final IDiscardableCutoffPredicate m_discardablePredicate;
028
029    public ClipAreasSettings(GeomDeviation cs2ct, GeomDeviation cwct, GeomDeviation rint, double discard_cutoffs, IDiscardableCutoffPredicate discard_pred) {
030        m_connectSubjectToClipTolerance = cs2ct;
031        m_clipperWayCleanupsTolerance = cwct;
032        m_reconnectIntersectionNodesTolerance = rint;
033        m_discardCutoffsPercent = discard_cutoffs;
034        m_discardablePredicate = discard_pred;
035    }
036
037    public ClipAreasSettings(GeomDeviation tolerance) {
038        this (tolerance, tolerance, defaultReconnectIntersectionNodesTolerance(), 0.0, null);
039    }
040
041    public ClipAreasSettings(GeomDeviation tolerance, double discard_cutoffs, IDiscardableCutoffPredicate discard_pred) {
042        this (tolerance, tolerance, defaultReconnectIntersectionNodesTolerance(), discard_cutoffs, discard_pred);
043    }
044
045    public GeomDeviation connectSubjectToClipTolerance() {
046        return m_connectSubjectToClipTolerance;
047    }
048
049    public GeomDeviation clipperWayCleanupsTolerance() {
050        return m_clipperWayCleanupsTolerance;
051    }
052
053    public GeomDeviation reconnectIntersectionNodesTolerance() {
054        return m_reconnectIntersectionNodesTolerance;
055    }
056
057    public double discardCutoffsPercent() {
058        return m_discardCutoffsPercent;
059    }
060
061    public IDiscardableCutoffPredicate discardablePredicate() {
062        return m_discardablePredicate;
063    }
064
065    public static GeomDeviation defaultReconnectIntersectionNodesTolerance() {
066        return new GeomDeviation(0.01, Math.PI / 40);
067    }
068}

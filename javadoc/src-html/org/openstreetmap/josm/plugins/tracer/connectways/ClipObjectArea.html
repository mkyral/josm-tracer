031    private final WayEditor m_editor;
032    private final ClipAreasSettings m_settings;
033    private final PostTraceNotifications m_postTraceNotifications;
034
035    public ClipObjectArea (WayEditor editor, ClipAreasSettings settings, PostTraceNotifications notifications) {
036        m_editor = editor;
037        m_settings = settings;
038        m_postTraceNotifications = notifications;
039    }
040
041    public EdObject clipObject(EdObject subject, AreaPredicate filter, LatLon anchor) {
042
043        // Note: only new objects with a proper tagging are clipped now. The algorithms below are too
044        // primitive and agressive for modification of existing OSM objects...
045
046        if (subject.hasOriginal()) {
047            addPostTraceNotification(tr("Subject way has original object, object clip is not supported yet."));
048            return subject;
049        }
050
051        if (subject.isMultipolygon()) {
052            EdMultipolygon mp = (EdMultipolygon) subject;
053            if (mp.containsTaggedWays()) {
054                addPostTraceNotification(tr("Subject multipolygon contains tagged ways, object clip is not supported yet."));
055                return subject;
056            }
057            for (EdWay way: mp.allWays()) {
058                if (way.hasOriginal ()) {
059                    addPostTraceNotification(tr("Subject way has original object, object clip is not supported yet."));
060                    return subject;
061                }
062            }
063        }
064        else if (!subject.isWay ()) {
065            throw new AssertionError ("EdObject is not a way or multipolygon");
066        }
067
068        Set<EdObject> areas1 = m_editor.useAllAreasInBBox(subject.getBBox(), filter);
069        Set<EdObject> areas = new HashSet<> ();
070
071        // (Subject must always be removed from clips set before clipping starts,
072        // because clipSubject operations transform the subject into new EdObject instances.)
073        for (EdObject obj : areas1) {
074            if (obj == subject)
075                continue;
076            if (obj.isMultipolygon() && subject.isWay() && ((EdMultipolygon)obj).containsWay((EdWay)subject))
077                continue;
078            areas.add (obj);
079        }
080
081        for (EdObject obj : areas) {
082            subject = clipSubject (obj, subject, anchor);
083        }
084
085        return subject;
086    }
087
088    private EdObject clipSubject (EdObject clip, EdObject subject, LatLon anchor) {
089
090        System.out.println ("Clipping subject id=" + Long.toString (subject.getUniqueId()) + " by clip id=" + Long.toString (clip.getUniqueId()));
091
092        AngPolygonClipper clipper = new AngPolygonClipper(m_editor, m_settings.clipperWayCleanupsTolerance(), m_settings.discardCutoffsPercent());
093        clipper.polygonDifference(clip, subject);
094        if (clipper.changesOutsideDataBounds()) {
095            addPostTraceNotification(tr("Subject {0} would be modified outside downloaded area, ignoring.", subject.getUniqueId()));
096            return subject;
097        }
098
099        List<List<EdNode>> outers = clipper.outerPolygons();
100        List<List<EdNode>> inners = clipper.innerPolygons();
101
102        if (outers.isEmpty() && inners.isEmpty()) {
103            // Never completely discard whole subject
104            System.out.println ("Subject would be removed...");
105            return subject;
106        }
107
108        else if (outers.size() == 1 && inners.isEmpty()) {
109            return handleSimpleResult(clip, subject, outers.get(0), anchor);
110        } else if ((outers.size() + inners.size()) > 1) {
111            return handleMultiResult(clip, subject, outers, inners, anchor);
112        } else {
113            throw new AssertionError(tr("PolygonClipper.polygonDifference returned nonsense!"));
114        }
115    }
116
117    private void addPostTraceNotification(String msg) {
118        if (m_postTraceNotifications == null)
119            return;
120        m_postTraceNotifications.add(msg);
121    }
122
123    private EdObject handleSimpleResult(EdObject clip, EdObject subject, List<EdNode> way, LatLon anchor) {
124
125        // Check anchor point, if specified
126        if (anchor != null && !isPointInsideWay (way, anchor)) {
127            return subject;
128        }
129
130        // Way -> Way
131        if (subject.isWay()) {
132            EdWay wsubj = (EdWay) subject;
133            wsubj.setNodes (way);
134            connectSubjectIntersectionsToClip (clip, wsubj);
135            return wsubj;
136        }
137
138        // Multipolygon -> Way
139        if (subject.isMultipolygon()) {
140            EdMultipolygon mpsubj = (EdMultipolygon) subject;
141            if (mpsubj.containsTaggedWays())
142                return mpsubj;
143            EdWay wsubj = m_editor.newWay(way);
144            Map<String, String> keys = mpsubj.getKeys();
145            keys.remove("type");
146            wsubj.setKeys(keys);
147            mpsubj.deleteShallow();
148            connectSubjectIntersectionsToClip (clip, wsubj);
149            return wsubj;
150        }
151
152        throw new AssertionError ("EdObject is not a way or multipolygon");
153    }
154
155    private EdObject handleMultiResult(EdObject clip, EdObject subject, List<List<EdNode>> outers, List<List<EdNode>> inners, LatLon anchor) {
156
157        // Check anchor point, if specified
158        if (anchor != null && !isPointInsideMultipolygon (outers, inners, anchor)) {
159            return subject;
160        }
161
162        // Way -> Multipolygon
163        if (subject.isWay ()) {
164            EdMultipolygon mpsubj = m_editor.newMultipolygon();
165            mpsubj.setKeys(subject.getKeys());
166            subject.deleteShallow();
167            for (List<EdNode> outer: outers) {
168                mpsubj.addOuterWay(m_editor.newWay (outer));
169            }
170            for (List<EdNode> inner: inners) {
171                mpsubj.addInnerWay(m_editor.newWay (inner));
172            }
173            connectSubjectIntersectionsToClip (clip, mpsubj);
174            return mpsubj;
175        }
176
177        // Multipolygon -> Multipolygon
178        if (subject.isMultipolygon()) {
179            EdMultipolygon mpsubj = (EdMultipolygon)subject;
180            mpsubj.removeAllWays();
181            for (List<EdNode> outer: outers) {
182                mpsubj.addOuterWay(m_editor.newWay (outer));
183            }
184            for (List<EdNode> inner: inners) {
185                mpsubj.addInnerWay(m_editor.newWay (inner));
186            }
187            connectSubjectIntersectionsToClip (clip, mpsubj);
188            return mpsubj;
189        }
190
191        throw new AssertionError ("EdObject is not a way or multipolygon");
192    }
193
194    private void connectSubjectIntersectionsToClip (EdObject clip, EdObject newsubject) {
195        // Connect clip to subject, this step guarantees that all newly created
196        // intersection nodes will be included in both ways.
197        clip.connectNonIncludedTouchingNodes(m_settings.reconnectIntersectionNodesTolerance(), newsubject);
198    }
199

201        // #### fixme
202        return true;
203    }
204
205    private boolean isPointInsideMultipolygon(List<List<EdNode>> outers, List<List<EdNode>> inners, LatLon anchor) {
206        // #### fixme
207        return true;
208    }

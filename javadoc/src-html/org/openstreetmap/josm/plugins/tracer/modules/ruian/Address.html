172    public String getHouseNumber() {
173      String housenumber;
174      if (m_conscriptionnumber.length() > 0 && m_streetnumber.length() > 0) {
175        housenumber = m_conscriptionnumber + "/" + m_streetnumber;
176      } else if (m_provisionalnumber.length() > 0 && m_streetnumber.length() > 0) {
177        housenumber = "ev." + m_provisionalnumber + "/" + m_streetnumber;
178      } else if (m_conscriptionnumber.length() > 0) {
179        housenumber = m_conscriptionnumber;
180      } else if (m_provisionalnumber.length() > 0) {
181        housenumber = "ev." + m_provisionalnumber;
182      } else {
183        housenumber = "";
184      }
185
186      return housenumber;
187    }
